59 CANopen::Driver::activate_PDO(PDOFunctionCode fn, bool set) {
60  uint16_t index = 0;
61  std::string reg_name;
62 
63  if((fn & 0x80) == 0x00) //R_PDO
64  {
65  index = (0x1400 + (fn >> 8) - 2);
66  reg_name = "R_PDO" + std::to_string((fn>>8)-1);
67  } else {
68  index = (0x1800 + (fn >> 8) - 1);
69  reg_name = "T_PDO" + std::to_string(fn>>8);
70  }
71  reg_name+="_setting";
72 
73  Register reg = (Register)( (index<<16) + 0x0001);
74  int32_t val = (set ? 0x04000000 : 0x80000000) + fn + m_node_id;
75  if(m_parameters.count(reg)==0)
76  {
77  m_parameters[reg] = new Parameter(reg_name, val, index, 1);
78  this->send(m_parameters[reg]);
79  while(m_parameters[reg]->sdo_flag.test_and_set());
80 
81  IF_VERBOSE(1, std::cout << reg_name << " " << ((set)?"activated":"desactivated") << "\n", m_verbose_level)
82 
83  }
84  else if((int32_t)*m_parameters[reg] != val)
85  {
86  *m_parameters[reg] = val;
87  this->send(m_parameters[reg]);
88  while(m_parameters[reg]->sdo_flag.test_and_set());
89 
90  IF_VERBOSE(1, std::cout << reg_name << " " << ((set)?"activated":"desactivated") << "\n", m_verbose_level)
91  }

125 void
126 CANopen::Driver::T_socket() {
127  //create a socket dedicated to the incomming pdo (T_PDO)
128  Socket socket(m_ifname, false);
129  //set the filter to only receive T_PDO messages
130  socket.add_filter({{m_node_id,0x07f}}); //all message with the right node id
131 
132  std::shared_ptr<Message> msg;
133  Payload p;
134  Message::FunctionCode fn;
135 
136  IF_VERBOSE(1, std::cout << "Receiving socket thread ready.\n";, m_verbose_level)
137 
138 
139  t_socket_flag.clear();
140  while(1) {
141  //try to receive pdo
142  msg = socket.receive();
143  if(msg.get() != nullptr) { //if pdo received
144  fn = msg->function_code();
145  p = msg->payload();
146  switch(fn) {
147  case CANopen::Message::Emergency:
148  {
149  std::shared_ptr<CANopen::EMCYMessage> emcy_msg = std::dynamic_pointer_cast<CANopen::EMCYMessage>(msg);
150  std::cout << "Error {" << (int)emcy_msg->node_id() << "} code: "<< std::hex << (int)emcy_msg->code() << " reg:" << (int)emcy_msg->reg() << "\n";
151  }
152  break;
153  case CANopen::Message::TimeStamp:
154  break;
155  case CANopen::Message::PDO1Transmit:
156  case CANopen::Message::PDO2Transmit:
157  case CANopen::Message::PDO3Transmit:
158  case CANopen::Message::PDO4Transmit:
159  if(m_PDO_map.count((PDOFunctionCode)fn)) {
160 
161  for(auto param : m_PDO_map[(PDOFunctionCode)fn])
162  {
163  if(param->from_payload(p, param->pdo_slot))
164  param->callback();
165  //if(param == m_parameters[DCOMcontrol])
166  //std::cout << param->name << ": "<< std::hex << (uint16_t)*param << " " << p << "\n";
167  }
168  }
169  break;
170  case CANopen::Message::SDOTransmit:
171  if(m_parameters.count((Register)msg->id()))
172  {
173  std::shared_ptr<CANopen::SDOMessage> sdo_msg = std::dynamic_pointer_cast<CANopen::SDOMessage>(msg);
174  if(!sdo_msg->is_confirmation() && !sdo_msg->is_error())
175  m_parameters[(Register)msg->id()]->from_payload(p);
176  m_parameters[(Register)msg->id()]->sdo_flag.clear();
177  }
178  break;
179  default:
180  break;
181  }
182  }
183  }

187 CANopen::Driver::RPDO_socket() {
188  //create a socket dedicated to the incomming pdo (T_PDO)
189  Socket socket(m_ifname, false);
190  PDOFunctionCode RPDO_fn_list[] = {PDO1Receive, PDO2Receive, PDO3Receive, PDO4Receive};
191 
192 
193  IF_VERBOSE(1, std::cout << "Sending PDO socket thread ready.\n";, m_verbose_level)
194 
195  rpdo_socket_flag.clear();
196  while(1)
197  {
198  for(auto fn : RPDO_fn_list) {
199  if(m_PDO_map.count(fn)) {
200  bool should_update=false, should_update_PDO1=false;
201  Payload payload;
202 
203  for(auto p : m_PDO_map[fn])// for each parameter mapped on the pdo, get the payload and store it at the right positon + get if the data has to be updated.
204  {
205  //std::cout << p->name << "\n";
206  payload.store_at(p->payload(((p != m_parameters[DCOMcontrol])?&should_update:&should_update_PDO1)), p->pdo_slot);
207  }
208 
209  if(should_update)
210  {
211  try {
212  socket.send(PDOMessage((PDOMessage::PDOFunctionCode) fn, m_node_id, payload));
213  }
214  catch (const std::runtime_error& e) {
215  IF_VERBOSE(1, std::cerr << e.what();, m_verbose_level)
216  }
217 
218  }
219  else if(should_update_PDO1)
220  {
221  payload.resize(m_parameters[DCOMcontrol]->size);
222  try {
223  socket.send(PDOMessage((PDOMessage::PDOFunctionCode) PDO1Receive, m_node_id, payload));
224  }
225  catch (const std::runtime_error& e) {
226  IF_VERBOSE(1, std::cerr << e.what();, m_verbose_level)
227  }

229  }
